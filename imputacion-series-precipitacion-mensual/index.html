Imputacion matricial de series de precipitacion mensual

Este notebook formula el problema como una matriz completa tiempo x estaciones en lugar de trabajar una estacion por vez. El objetivo es evaluar, con un protocolo reproducible y sin fuga de informacion, que combinacion entre reconstruccion inicial e imputacion final reconstruye mejor valores reales ocultados artificialmente.

La narrativa del notebook sigue tres capas:

  1. Construccion del escenario de evaluacion

    • matriz original con faltantes reales,

    • mascara artificial tipo overlay,

    • benchmark externo separado.

  2. Reconstruccion inicial de la matriz

    • interpolacion por estacion,

    • forecast_filled por pronostico recursivo y backcasting.

  3. Imputacion final multivariada

    • MLP,

    • LSTM,

    • CNN1D.

La validacion principal usa valores realmente observados que fueron ocultados artificialmente. La hoja LluviaAcum7615_Reconstruida se conserva solo como benchmark externo de una metodologia matricial previa.

Paso 3. Reconstruccion inicial de toda la matriz¶

Antes de entrenar modelos, generamos dos versiones base de la matriz de trabajo. Estas reconstrucciones iniciales no son el resultado final, sino el punto de partida sobre el cual se construyen ventanas y se entrena cada arquitectura.

Se prueban dos estrategias de pre-relleno por estacion:

  • interpolacion

  • forecast_filled con pronostico recursivo estacional y backcasting

def fill_with_recursive_forecast(series: pd.Series, season_length: int = 12, local_window: int = 3) -> pd.Series:
    values = pd.to_numeric(series, errors='coerce').to_numpy(copy=True, dtype=float)
    index = series.index
    month_means = series.groupby(series.index.month).mean()
    global_mean = float(np.nanmean(values)) if np.isfinite(np.nanmean(values)) else 0.0

    def fallback(month):
        val = month_means.get(month, np.nan)
        return float(val) if pd.notna(val) else global_mean

    for i in range(len(values)):
        if np.isnan(values[i]):
            candidates = []
            if i - season_length >= 0 and not np.isnan(values[i - season_length]):
                candidates.append(values[i - season_length])
            prev_vals = [values[j] for j in range(max(0, i - local_window), i) if not np.isnan(values[j])]
            if prev_vals:
                candidates.append(float(np.mean(prev_vals)))
            candidates.append(fallback(index[i].month))
            values[i] = float(np.mean(candidates))

    for i in range(len(values) - 1, -1, -1):
        if np.isnan(values[i]):
            candidates = []
            if i + season_length < len(values) and not np.isnan(values[i + season_length]):
                candidates.append(values[i + season_length])
            next_vals = [values[j] for j in range(i + 1, min(len(values), i + 1 + local_window)) if not np.isnan(values[j])]
            if next_vals:
                candidates.append(float(np.mean(next_vals)))
            candidates.append(fallback(index[i].month))
            values[i] = float(np.mean(candidates))

    return pd.Series(values, index=index, name=series.name)

from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score


def nash_sutcliffe(y_true, y_pred):
    y_true = np.asarray(y_true).reshape(-1)
    y_pred = np.asarray(y_pred).reshape(-1)
    num = np.sum((y_true - y_pred) ** 2)
    den = np.sum((y_true - np.mean(y_true)) ** 2)
    return 1 - num / den if den != 0 else -np.inf


def compute_metrics(y_true, y_pred):
    y_true = np.asarray(y_true).reshape(-1)
    y_pred = np.asarray(y_pred).reshape(-1)
    mse = mean_squared_error(y_true, y_pred)
    return {
        'MSE': mse,
        'RMSE': np.sqrt(mse),
        'MAE': mean_absolute_error(y_true, y_pred),
        'R2': r2_score(y_true, y_pred),
        'NSE': nash_sutcliffe(y_true, y_pred),
    }

def build_windows_from_matrix(Y_base: pd.DataFrame, Y_true: pd.DataFrame, mask_valid_targets: pd.DataFrame, window_size: int = 12):
    X_list, y_list, mask_list, t_list = [], [], [], []
    Yb = Y_base.values.astype(float)
    Yt = Y_true.values.astype(float)
    Mv = mask_valid_targets.values.astype(bool)
    for t in range(window_size, len(Y_base)):
        X_list.append(Yb[t-window_size:t, :])
        y_list.append(np.nan_to_num(Yt[t, :], nan=0.0))
        mask_list.append(Mv[t, :].astype(float))
        t_list.append(t)
    return np.array(X_list), np.array(y_list), np.array(mask_list), np.array(t_list)

Paso 9. Escalado matricial por estacion¶

Escalamos la matriz por variable antes del entrenamiento. Este paso se aplica despues de definir la mascara y despues de construir las series base, para mantener el orden metodologico y reducir riesgo de fuga de informacion.

from sklearn.preprocessing import StandardScaler


def scale_multivariate_windows(X_train, X_test, y_train, y_test):
    n_features = X_train.shape[2]
    scaler_x = StandardScaler()
    scaler_y = StandardScaler()

    X_train_2d = X_train.reshape(-1, n_features)
    X_test_2d = X_test.reshape(-1, n_features)
    X_train_scaled = scaler_x.fit_transform(X_train_2d).reshape(X_train.shape)
    X_test_scaled = scaler_x.transform(X_test_2d).reshape(X_test.shape)

    y_train_scaled = scaler_y.fit_transform(y_train)
    y_test_scaled = scaler_y.transform(y_test)
    return X_train_scaled, X_test_scaled, y_train_scaled, y_test_scaled, scaler_x, scaler_y

Paso 10. Definir perdida enmascarada y utilidades de entrenamiento¶

Como la salida es multivariada y no todas las posiciones son igualmente validas para entrenar, usamos una perdida enmascarada que ignora objetivos no observados y evalua solo donde hay informacion confiable.

import tensorflow as tf
from tensorflow.keras.callbacks import EarlyStopping
from tensorflow.keras.layers import Dense, Input, LSTM, Conv1D, Flatten, GlobalAveragePooling1D
from tensorflow.keras.models import Sequential

tf.get_logger().setLevel('ERROR')

n_stations = Y_raw.shape[1]


def pack_targets(y, mask):
    return np.concatenate([y, mask], axis=1)


def masked_mse_loss(n_outputs):
    def loss(y_true_masked, y_pred):
        y_true = y_true_masked[:, :n_outputs]
        mask = y_true_masked[:, n_outputs:]
        sq = tf.square(y_true - y_pred) * mask
        denom = tf.reduce_sum(mask, axis=1)
        denom = tf.where(denom == 0, tf.ones_like(denom), denom)
        return tf.reduce_mean(tf.reduce_sum(sq, axis=1) / denom)
    return loss


def masked_metrics(y_true_masked, y_pred):
    y_true = y_true_masked[:, :n_stations]
    mask = y_true_masked[:, n_stations:]
    y_true_flat = y_true[mask > 0]
    y_pred_flat = y_pred[mask > 0]
    return compute_metrics(y_true_flat, y_pred_flat)


def graficar_perdida(history, titulo='Perdida: Train vs Val', figsize=(10, 4)):
    plt.figure(figsize=figsize)
    plt.plot(history.history['loss'], label='Train loss', linewidth=2)
    plt.plot(history.history['val_loss'], label='Val loss', alpha=0.7)
    plt.title(titulo)
    plt.xlabel('Epocas')
    plt.ylabel('Valor de la perdida')
    plt.legend()
    plt.grid(True, linestyle='--', alpha=0.5)
    plt.show()


def build_mlp_multivariate(window_size, n_stations):
    model = Sequential([
        Input(shape=(window_size, n_stations)),
        Flatten(),
        Dense(64, activation='relu'),
        Dense(n_stations)
    ])
    model.compile(optimizer='adam', loss=masked_mse_loss(n_stations))
    return model


def build_lstm_multivariate(window_size, n_stations):
    model = Sequential([
        Input(shape=(window_size, n_stations)),
        LSTM(16, activation='tanh'),
        Dense(n_stations)
    ])
    model.compile(optimizer='adam', loss=masked_mse_loss(n_stations))
    return model


def build_cnn_multivariate(window_size, n_stations):
    model = Sequential([
        Input(shape=(window_size, n_stations)),
        Conv1D(filters=8, kernel_size=3, activation='relu'),
        GlobalAveragePooling1D(),
        Dense(n_stations)
    ])
    model.compile(optimizer='adam', loss=masked_mse_loss(n_stations))
    return model


def train_and_validate(build_fn, X_train, y_train_pack, X_test, y_test_pack, title, epochs=40, batch_size=8):
    tf.random.set_seed(42)
    np.random.seed(42)
    model = build_fn(window_size, n_stations)
    early = EarlyStopping(monitor='val_loss', patience=8, restore_best_weights=True)
    history = model.fit(
        X_train,
        y_train_pack,
        validation_data=(X_test, y_test_pack),
        epochs=epochs,
        batch_size=batch_size,
        verbose=0,
        callbacks=[early],
    )
    graficar_perdida(history, titulo=title)
    y_pred_test = model.predict(X_test, verbose=0)
    test_metrics = masked_metrics(y_test_pack, y_pred_test)
    print(title, test_metrics)
    return model, history, test_metrics

Paso 12. Imputacion final sobre la matriz de trabajo¶

artificial_eval_rows, artificial_eval_cols = np.where(M_art)

def multivariate_recursive_impute(model, Y_base, scaler_x, scaler_y):
    Y_imp = Y_base.to_numpy(copy=True, dtype=float)
    missing_mask = M_work.copy()
    for t in range(window_size, len(Y_imp)):
        if not missing_mask[t].any():
            continue
        X_window = Y_imp[t-window_size:t, :]
        X_scaled = scaler_x.transform(X_window.reshape(-1, n_stations)).reshape(1, window_size, n_stations)
        pred_scaled = model.predict(X_scaled, verbose=0)
        pred = scaler_y.inverse_transform(pred_scaled).reshape(-1)
        Y_imp[t, missing_mask[t]] = pred[missing_mask[t]]
    return Y_imp
